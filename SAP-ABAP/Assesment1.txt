DATA V_TOTAL TYPE P.
PARAMETER P_AMOUNT TYPE P.
PARAMETER P_PROFIT_PERCENT TYPE P.
PARAMETER P_DAYS TYPE I.
PARAMETER BOX AS CHECKBOX.

IF BOX EQ 'X'.
  V_TOTAL = P_AMOUNT * P_DAYS * (1 + (P_PROFIT_PERCENT / 100)).
ELSE.
  V_TOTAL = P_AMOUNT.
  DO P_DAYS TIMES.
    V_TOTAL = V_TOTAL * (1 + (P_PROFIT_PERCENT / 100)).
  ENDDO.
ENDIF.

WRITE: / 'P_AMOUNT:', P_AMOUNT,
       / 'P_DAYS:', P_DAYS,
       / 'V_TOTAL:', V_TOTAL.
